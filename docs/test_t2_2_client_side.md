# T2.2 Client-Side Testing Guide: Cross-User Data Isolation

## Overview
This guide provides comprehensive client-side tests to verify that all tables properly prevent cross-user data access using the Supabase client with anon key.

## Test Setup

### 1. Create Test Users
Create two test users in your application:

```typescript
// Test users for T2.2
```

### 2. Test Data Setup

### Test 1: Bookmarks Table Isolation

```typescript
// test-bookmarks-isolation.ts
import { createClient } from '@supabase/supabase-js';

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!
);

async function testBookmarksIsolation() {
  console.log('üîñ Testing Bookmarks Table Isolation...');

  // Sign in as User 1
  const { data: user1Auth } = await supabase.auth.signInWithPassword({
    email: 'testuser1-t2@example.com',
    password: 'testpassword123'
  });

  // Create a bookmark as User 1
  const { data: bookmark1, error: createError } = await supabase
    .from('bookmarks')
    .insert({
      url: 'https://example.com/user1-bookmark',
      title: 'User 1 Test Bookmark',
      description: 'This should only be visible to User 1'
    })
    .select()
    .single();

  if (createError) {
    console.error('‚ùå Failed to create bookmark:', createError);
    return;
  }

  console.log('‚úÖ User 1 created bookmark:', bookmark1.id);

  // Sign out and sign in as User 2
  await supabase.auth.signOut();
  const { data: user2Auth } = await supabase.auth.signInWithPassword({
    email: 'testuser2-t2@example.com',
    password: 'testpassword123'
  });

  // Try to access User 1's bookmark
  const { data: bookmarks, error } = await supabase
    .from('bookmarks')
    .select('*')
    .eq('id', bookmark1.id);

  if (error) {
    console.error('‚ùå Error accessing bookmarks:', error);
  } else if (bookmarks.length === 0) {
    console.log('‚úÖ SUCCESS: User 2 cannot see User 1\'s bookmark');
  } else {
    console.log('‚ùå FAILURE: User 2 can see User 1\'s bookmark');
  }

  // Try to update User 1's bookmark
  const { error: updateError } = await supabase
    .from('bookmarks')
    .update({ title: 'Hacked by User 2' })
    .eq('id', bookmark1.id);

  if (updateError) {
    console.log('‚úÖ SUCCESS: User 2 blocked from updating User 1\'s bookmark');
  } else {
    console.log('‚ùå FAILURE: User 2 was able to update User 1\'s bookmark');
  }

  // Try to delete User 1's bookmark
  const { error: deleteError } = await supabase
    .from('bookmarks')
    .delete()
    .eq('id', bookmark1.id);

  if (deleteError) {
    console.log('‚úÖ SUCCESS: User 2 blocked from deleting User 1\'s bookmark');
  } else {
    console.log('‚ùå FAILURE: User 2 was able to delete User 1\'s bookmark');
  }

  await supabase.auth.signOut();
}

testBookmarksIsolation();
```

```typescript
import { createClient } from '@supabase/supabase-js';

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
```

```typescript
    .single();

  const { data: bookmark } = await supabase
    .from('bookmarks')

    .single();

  const { data: collectionItem, error: createError } = await supabase
    .from('collection_items')

      collection_id: collection.id,

    console.error('‚ùå Failed to create collection item:', createError);
    return;
  }

  console.log('‚úÖ User 1 created collection item');

  await supabase.auth.signOut();
  await supabase.auth.signInWithPassword({
    email: 'testuser2-t2@example.com',
    password: 'testpassword123'
  });

  // Try to access User 1's collection items
  const { data: collectionItems, error } = await supabase
    .from('collection_items')
    .select('*')
    .eq('collection_id', collection.id);

  if (error) {
    console.error('‚ùå Error accessing collection items:', error);

    console.log('‚úÖ SUCCESS: User 2 cannot see User 1\'s collection items');

    console.log('‚ùå FAILURE: User 2 can see User 1\'s collection items');

      collection_id: collection.id,
```

### Test 6: Public Sharing Functionality

```typescript
// test-public-sharing.ts
import { createClient } from '@supabase/supabase-js';

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!
);

async function testPublicSharing() {
  console.log('üåê Testing Public Sharing Functionality...');

  // Sign in as User 1
  await supabase.auth.signInWithPassword({
    email: 'testuser1-t2@example.com',
    password: 'testpassword123'
  });

  // Create a public bookmark
  const { data: publicBookmark } = await supabase
    .from('bookmarks')
    .insert({
      url: 'https://public-example.com',
      title: 'Public Bookmark',
      is_public: true
    })
    .select()
    .single();

  // Create a public collection
  const { data: publicCollection } = await supabase
    .from('collections')
    .insert({
      name: 'Public Collection',
      type: 'manual',
      is_public: true
    })
    .select()
    .single();

  console.log('‚úÖ User 1 created public bookmark and collection');

  // Sign out and sign in as User 2
  await supabase.auth.signOut();
  await supabase.auth.signInWithPassword({
    email: 'testuser2-t2@example.com',
    password: 'testpassword123'
  });

  // User 2 should be able to see public items
  const { data: publicBookmarks } = await supabase
    .from('bookmarks')
    .select('*')
    .eq('is_public', true);

  const { data: publicCollections } = await supabase
    .from('collections')
    .select('*')
    .eq('is_public', true);

  console.log(`‚úÖ User 2 can see ${publicBookmarks.length} public bookmarks`);
  console.log(`‚úÖ User 2 can see ${publicCollections.length} public collections`);

  // Sign out completely
  await supabase.auth.signOut();

  // Unauthenticated users should also see public items
  const { data: unauthBookmarks } = await supabase
    .from('bookmarks')
    .select('*')
    .eq('is_public', true);

  const { data: unauthCollections } = await supabase
    .from('collections')
    .select('*')
    .eq('is_public', true);

  console.log(`‚úÖ Unauthenticated users can see ${unauthBookmarks.length} public bookmarks`);
  console.log(`‚úÖ Unauthenticated users can see ${unauthCollections.length} public collections`);
}

testPublicSharing();
```
